비교 연산자

비교 연산자 예제 코드를 보여주세요 

# 변수 선언 및 초기화
a = 10
b = 20

# 동등 비교
print("a == b:", a == b)  # False

# 부등 비교
print("a != b:", a != b)  # True

# 크기 비교
print("a > b:", a > b)   # False
print("a < b:", a < b)   # True

# 크거나 같음, 작거나 같음
print("a >= b:", a >= b)  # False
print("a <= b:", a <= b)  # True
