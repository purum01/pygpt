Pandas 학습 가이드

DataFrame 생성, 파일 저장, 파일 읽기 
   
import pandas as pd

# 데이터 생성 (딕셔너리 형태)
data = {
    'Name': ['Alice', 'Bob', 'Charlie', 'David'],
    'Age': [25, 30, 35, 40],
    'City': ['New York', 'Los Angeles', 'Chicago', 'Houston'],
    'Score': [88.5, 92.3, 85.6, 90.1]
}

# 딕셔너리 데이터를 DataFrame으로 변환
doc = pd.DataFrame(data)

# DataFrame 출력
print(doc)    

# 처리된 데이터를 새로운 CSV 파일로 저장
doc.to_csv('data.csv', index=False)

# CSV 파일 불러오기
df = pd.read_csv('data.csv')

# 데이터 확인
print(df.head())
데이터 확인하기 
   
# 데이터의 첫 5행 출력
print(df.head())

# 데이터의 정보 출력
print(df.info())

# 각 열의 데이터 유형 확인
print(df.dtypes)
특정 열 선택 및 데이터 접근 
   
# 'Name' 열만 선택하여 출력
print(df['Name'])

# 여러 열 선택하기
print(df[['Name', 'City']])

# 특정 행의 데이터 선택 (loc를 이용한 인덱스 접근)
print(df.loc[0])  # 첫 번째 행 출력
Pandas 데이터 필터링 
   
# 나이가 30 이상인 사람들만 필터링
adults = df[df['Age'] >= 30]
print(adults)
데이터 정렬 
   
# 'Age' 열을 기준으로 오름차순 정렬
sorted_by_age = df.sort_values(by='Age', ascending=True)
print(sorted_by_age)

# 'Score' 열을 기준으로 내림차순 정렬
sorted_by_score = df.sort_values(by='Score', ascending=False)
print(sorted_by_score)
데이터 추가 및 수정 
   
# 새로운 열 추가
df['Occupation'] = ['Engineer', 'Doctor', 'Artist', 'Lawyer']
print(df)


# 새로운 행 추가할 데이터 생성
new_row = {'Name': 'Frank', 'Age': 28, 'City': 'San Francisco', 'Score': 87.0, 'Occupation': 'Lawyer'}
new_df =  pd.DataFrame([new_row])

# 기존 DataFrame과 새로운 행을 결합
df = pd.concat([df, new_df], ignore_index=True)
print(df)
데이터 요약 및 통계 분석 
   
# 각 열에 대한 기본 통계 요약 정보
print(df.describe())

# 'Score' 열의 평균 구하기
print(df['Score'].mean())

# 'Age' 열의 최댓값과 최솟값 구하기
print(df['Age'].max())
print(df['Age'].min())
  • count: 해당 열에 있는 값의 개수(결측값은 제외).
  • mean: 평균값.
  • std: 표준편차(데이터의 분산 정도).
  • min: 최소값.
  • 25%: 1사분위수 (데이터의 25%가 이 값 이하).
  • 50% (중앙값): 2사분위수 (데이터의 50%가 이 값 이하).
  • 75%: 3사분위수 (데이터의 75%가 이 값 이하).
  • max: 최대값.
DataFrame 병합 
   
import pandas as pd

# 첫 번째 DataFrame 생성
data1 = {
    'ID': [1, 2, 3, 4],
    'Name': ['Alice', 'Bob', 'Charlie', 'David'],
    'Age': [25, 30, 35, 40]
}
df1 = pd.DataFrame(data1)

# 두 번째 DataFrame 생성
data2 = {
    'ID': [3, 4, 5, 6],
    'City': ['Chicago', 'Houston', 'San Francisco', 'Los Angeles'],
    'Score': [88.5, 92.3, 85.6, 90.1]
}
df2 = pd.DataFrame(data2)

# 두 개의 DataFrame을 ID 열을 기준으로 병합
merged_df = pd.merge(df1, df2, on='ID', how='inner')
merged_df2 = pd.merge(df1, df2, on='ID', how='outer')

# 병합된 DataFrame 출력
print(merged_df)
print(merged_df2)
