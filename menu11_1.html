#조건
1. 로또번호는 1부터 45사이의 정수값임
2. 6개의 로또 번호 생성함
3. 생성된  6개 로또 번호 오름차순 정렬함
4. 6개의 로또번호를  5세트 생성함
답변 

import random  # 파이썬의 랜덤 모듈을 가져옵니다. 랜덤한 숫자를 생성하는 데 사용합니다.

# 로또 번호를 생성하는 함수
def generate_lotto_numbers():
    # 1부터 45 사이의 숫자 중에서 6개의 숫자를 랜덤으로 선택합니다.
    # 'sample' 함수는 중복 없이 6개의 숫자를 선택해 줍니다.
    lotto_numbers = random.sample(range(1, 46), 6)

    # 선택된 6개의 숫자를 오름차순으로 정렬합니다.
    lotto_numbers.sort()

    # 정렬된 번호를 반환합니다.
    return lotto_numbers

# 여러 세트의 로또 번호를 생성하는 함수
def generate_lotto_sets(num_sets=5):
    # 비어 있는 리스트를 준비하여 각 세트의 로또 번호를 저장합니다.
    lotto_sets = []

    # 원하는 세트 수(num_sets)만큼 반복해서 로또 번호를 생성합니다.
    for _ in range(num_sets):
        # 로또 번호 한 세트를 생성하여 리스트에 추가합니다.
        lotto_sets.append(generate_lotto_numbers())

    # 생성된 모든 로또 세트를 반환합니다.
    return lotto_sets

# 로또 번호 5세트를 생성합니다.
lotto_sets = generate_lotto_sets()

# 생성된 로또 번호 세트를 출력합니다.
for i, lotto_set in enumerate(lotto_sets, 1):
    print(f"Set {i}: {lotto_set}")
