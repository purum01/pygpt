def generate_lotto_numbers():
    # 1λ¶€ν„° 45 μ‚¬μ΄μ μ«μ μ¤‘μ—μ„ 6κ°μ μ«μλ¥Ό λλ¤μΌλ΅ μ„ νƒν•©λ‹λ‹¤.
    # 'sample' ν•¨μλ” μ¤‘λ³µ μ—†μ΄ 6κ°μ μ«μλ¥Ό μ„ νƒν•΄ μ¤λ‹λ‹¤.
    lotto_numbers = random.sample(range(1, 46), 6)

    # μ„ νƒλ 6κ°μ μ«μλ¥Ό μ¤λ¦„μ°¨μμΌλ΅ μ •λ ¬ν•©λ‹λ‹¤.
    lotto_numbers.sort()

    return lotto_numbers

def generate_lotto_sets(num_sets=5):

    lotto_sets = []

        lotto_sets.append(generate_lotto_numbers())

    return lotto_sets

# λ΅λ λ²νΈ 5μ„ΈνΈλ¥Ό μƒμ„±ν•©λ‹λ‹¤.
